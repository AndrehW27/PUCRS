Objeto por fábricas


    
        function criarPessoa(nome, idade) {
            return {
                nome,
                idade,
                falar() {
                    console.log(`Meu nome é ${nome} e tenho ${idade} anos.`);
                }
            };
        }
        const pessoa1 = criarPessoa("Maria", 30);
        const pessoa2 = criarPessoa("Carlos", 40);
    

    


    
Objeto por construtor
 
    

    
        function Pessoa(nome, idade) {
            this.nome = nome;
            this.idade = idade;
            this.falar = function() {
                console.log(`Meu nome é ${this.nome} e tenho ${this.idade} anos.`);
            };
        }
        
        const pessoa1 = new Pessoa("Ana", 28);

            }
        };
        
        const pessoa = Object.create(prototipoPessoa);
